 15class NetworkManager:
 16    """Gère les communications réseau avec le serveur"""
 17    def __init__(self, host, port):
 18        """Initialise le gestionnaire réseau avec les paramètres de connexion.
 19
 20        Args:
 21            host (str): L'adresse IP du serveur.
 22            port (int): Le port de connexion du serveur.
 23        """
 24        self.host = host
 25        self.port = port
 26        self.sock = None
 27        self.buffer = ""
 28        self.connected = False
 29        self.client_id = str(uuid.uuid4())[:8]
 30
 31    def connect(self):
 32        """Établit la connexion socket TCP avec le serveur"""
 33        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 34        self.sock.connect((self.host, self.port))
 35        self.connected = True
 36
 37    def send(self, data):
 38        """Envoie des données au serveur via la socket.
 39
 40        Args:
 41            data (str): Les données à envoyer au serveur.
 42        """
 43        self.sock.sendall(data.encode())
 44
 45    def wait_for_line(self):
 46        """Attend de recevoir une ligne complète du serveur.
 47
 48        Raises:
 49            Exception: Si la connexion est fermée par le serveur.
 50        """
 51        while "\n" not in self.buffer:
 52            data = self.sock.recv(1024).decode()
 53            if not data:
 54                raise Exception("Connexion fermée")
 55            self.buffer += data
 56
 57    def read_line(self):
 58        """Lit et retourne une ligne du buffer de réception.
 59
 60        Returns:
 61            str: La ligne lue, sans espaces en début/fin.
 62        """
 63        line, self.buffer = self.buffer.split("\n", 1)
 64        return line.strip()
 65
 66    def receive_data(self):
 67        """Reçoit les données disponibles du serveur de manière non-bloquante.
 68
 69        Met à jour le buffer interne avec les nouvelles données reçues.
 70        Met à jour l'état de connexion si le serveur ferme la connexion.
 71        """
 72        rlist, _, _ = select.select([self.sock], [], [], 0.1)
 73        if rlist:
 74            data = self.sock.recv(1024).decode()
 75            if not data:
 76                print("Connexion fermée par le serveur")
 77                self.connected = False
 78                return
 79            self.buffer += data
 80
 81    def close(self):
 82        """Ferme la connexion réseau"""
 83        self.sock.close()
 84        print("Connexion fermée")

 86class SharedInventory:
 87    """Inventaire partagé de l'équipe"""
 88    def __init__(self):
 89        """Initialise l'inventaire partagé de l'équipe avec les ressources et objectifs."""
 90        self.resources = {
 91            "linemate": 0, "deraumere": 0, "sibur": 0,
 92            "mendiane": 0, "phiras": 0, "thystame": 0
 93        }
 94        self.total_needed = {"linemate": 9, "deraumere": 8, "sibur": 10, "mendiane": 5, "phiras": 6, "thystame": 1}
 95
 96    def add_resources(self, resources):
 97        """Ajoute des ressources à l'inventaire partagé de l'équipe.
 98
 99        Args:
100            resources (dict): Dictionnaire des ressources à ajouter {nom_ressource: quantité}.
101        """
102        for resource, count in resources.items():
103            if resource in self.resources:
104                self.resources[resource] += count
105
106    def get_missing_resources(self):
107        """Calcule les ressources encore nécessaires pour atteindre les objectifs.
108
109        Returns:
110            dict: Dictionnaire des ressources manquantes {nom_ressource: quantité_manquante}.
111        """
112        missing = {}
113        for resource, needed in self.total_needed.items():
114            current = self.resources[resource]
115            if current < needed:
116                missing[resource] = needed - current
117        return missing
118
119    def has_enough_for_level_8(self):
120        """Vérifie si l'équipe possède suffisamment de ressources pour le niveau 8.
121
122        Returns:
123            bool: True si toutes les ressources nécessaires sont collectées, False sinon.
124        """
125        has_enough = len(self.get_missing_resources()) == 0
126        return has_enough
127
128    def is_resource_needed(self, resource):
129        """Vérifie si une ressource spécifique est encore nécessaire.
130
131        Args:
132            resource (str): Le nom de la ressource à vérifier.
133        Returns:
134            bool: True si la ressource est encore nécessaire, False sinon.
135        """
136        if resource not in self.total_needed:
137            return False
138        return self.resources[resource] < self.total_needed[resource]

329    """Gère l'état du jeu et les informations du joueur"""
330    def __init__(self):
331        """Initialise l'état du jeu avec les valeurs par défaut."""
332        self.slots = 0

345    def parse_inventory(self, line):
346        """Parse la réponse d'inventaire du serveur et met à jour l'inventaire local.
347
348        Args:
349            line (str): La ligne de réponse du serveur contenant l'inventaire.
350        """
351        line = line.strip("[]")
352        resources = ["linemate", "deraumere", "sibur", "mendiane", "phiras", "thystame"]
353        for r in resources:
354            self.inventory[r] = 0
355        self.inventory["food"] = 0
356        items = [item.strip() for item in line.split(",") if item.strip()]
357        for item in items:
358            parts = item.split()
359            if len(parts) == 2:
360                resource_name = parts[0]
361                count = int(parts[1])
362                if resource_name in self.inventory:
363                    self.inventory[resource_name] = count
364
365    def parse_look(self, response):
366        """Parse la réponse de look du serveur et retourne la vue structurée.
367
368        Args:
369            response (str): La réponse du serveur à la commande Look.
370
371        Returns:
372            list: Liste de dictionnaires représentant chaque case visible.
373        """
374        clean_str = response.strip("[]").split(",")
375        tiles = []
376        for tile_str in clean_str:
377            tile_str = tile_str.strip()
378            if not tile_str:
379                tiles.append({})
380                continue
381            counts = {}
382            objects = [obj.strip().rstrip(',') for obj in tile_str.split() if obj.strip()]
383            for obj in objects:
384                obj = obj.strip()
385                if obj:
386                    if obj in counts:
387                        counts[obj] += 1
388                    else:
389                        counts[obj] = 1
390            tiles.append(counts)
391        return tiles

428class MovementManager:
429    """Gère les déplacements et la recherche de ressources"""
430    def __init__(self, game_state):
431        """Initialise le gestionnaire de mouvement avec l'état du jeu.
432
433        Args:
434            game_state (GameState): L'état actuel du jeu.
435        """
436        self.game_state = game_state

438    def find_best_resource(self, resource):
439        """Trouve l'index de la case contenant le plus de la ressource spécifiée.
440
441        Args:
442            resource (str): Le nom de la ressource à chercher.
443
444        Returns:
445            int or None: L'index de la meilleure case, ou None si aucune trouvée.
446        """
447        best_index = None
448        best_count = 0
449        for i, tile in enumerate(self.game_state.view):
450            count = tile.get(resource, 0)
451            if count > best_count:
452                best_count = count
453                best_index = i
454        return best_index
455
456    def take_all_on_tile(self, resource, index_tile, command_queue):
457        """Ajoute les commandes pour prendre toutes les ressources d'un type sur une case.
458
459        Args:
460            resource (str): Le nom de la ressource à prendre.
461            index_tile (int): L'index de la case dans la vue.
462            command_queue (list): La file des commandes à exécuter.
463        """
464        if index_tile >= len(self.game_state.view):
465            return
466        tile = self.game_state.view[index_tile]
467        amount = tile.get(resource, 0)
468        for _ in range(amount):
469            command_queue.append(f"Take {resource}")
470
471    def move_to(self, index, command_queue):
472        """Ajoute les commandes pour se déplacer vers une case spécifique de la vue.
473
474        Collecte automatiquement la nourriture rencontrée sur le chemin.
475
476        Args:
477            index (int): L'index de la case de destination dans la vue.
478            command_queue (list): La file des commandes à exécuter.
479        """
480        path_map = {
481            0: ([], [0]),
482            1: (["Forward", "Left", "Forward"], [0, 2, 1]),
483            2: (["Forward"], [0, 2]),
484            3: (["Forward", "Right", "Forward"], [0, 2, 3]),
485            4: (["Forward", "Forward", "Left", "Forward", "Forward"], [0, 2, 6, 5, 4]),
486            5: (["Forward", "Forward", "Left", "Forward"], [0, 2, 6, 5]),
487            6: (["Forward", "Forward"], [0, 2, 6]),
488            7: (["Forward", "Forward", "Right", "Forward"], [0, 2, 6, 7]),
489            8: (["Forward", "Forward", "Right", "Forward", "Forward"], [0, 2, 6, 7, 8])
490        }
491        moves, traversed_tiles = path_map.get(index, (["Forward"], [0, index]))
492        if len(traversed_tiles) > 0 and traversed_tiles[0] < len(self.game_state.view):
493            tile = self.game_state.view[traversed_tiles[0]]
494            if "food" in tile and tile["food"] > 0:
495                self.take_all_on_tile("food", traversed_tiles[0], command_queue)
496        tile_index_counter = 1
497        for cmd in moves:
498            command_queue.append(cmd)
499            if cmd == "Forward":
500                if tile_index_counter < len(traversed_tiles):
501                    tile_index = traversed_tiles[tile_index_counter]
502                    if tile_index < len(self.game_state.view):
503                        tile = self.game_state.view[tile_index]
504                        if "food" in tile and tile["food"] > 0:
505                            self.take_all_on_tile("food", tile_index, command_queue)
506                    tile_index_counter += 1
507        self.game_state.need_look = True
508
509    def move_towards_direction(self, direction, command_queue):
510        """Ajoute les commandes pour se déplacer dans une direction spécifique.
511
512        Args:
513            direction (int): La direction (0-8) vers laquelle se déplacer.
514            command_queue (list): La file des commandes à exécuter.
515        """
516        direction_moves = {
517            0: [],
518            1: ["Forward"],
519            2: ["Forward", "Left", "Forward"],
520            3: ["Left", "Forward"],
521            4: ["Left", "Forward", "Left", "Forward"],
522            5: ["Right", "Right", "Forward"],
523            6: ["Right", "Forward", "Right", "Forward"],
524            7: ["Right", "Forward"],
525            8: ["Forward", "Right", "Forward"]
526        }
527        moves = direction_moves.get(direction, [])
528        for move in moves:
529            command_queue.append(move)
530        if moves:
531            self.game_state.need_look = True
532
533    def search_priority_resources(self, shared_inventory, command_queue):
534        """Recherche les ressources manquantes par ordre de priorité.
535
536        Args:
537            shared_inventory (SharedInventory): L'inventaire partagé de l'équipe.
538            command_queue (list): La file des commandes à exécuter.
539
540        Returns:
541            bool: True si une ressource a été trouvée et ciblée, False sinon.
542        """
543        missing = shared_inventory.get_missing_resources()
544        if not missing:
545            return False
546        priority_order = ["thystame", "phiras", "mendiane", "sibur", "deraumere", "linemate"]
547        for resource in priority_order:
548            if resource in missing:
549                index = self.find_best_resource(resource)
550                if index is not None and index < len(self.game_state.view):
551                    if index == 0:
552                        self.take_all_on_tile(resource, 0, command_queue)
553                    else:
554                        self.move_to(index, command_queue)
555                    return True
556        return False
